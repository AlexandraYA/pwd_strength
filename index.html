  • javascript
    $(document).ready(function() {
     $('#protect').html("ненадежный пароль");
     var terms = {'length': false, 'az': false, 'AZ': false,
                  'symbols': false, 'int': false};
     var protect;
     var error;

     $('#password').keyup(function() {
      var pwd = $(this).val();
      protect = 0;
      //проверяем пароль на длину >=6 символов
      if ( (pwd.length > 6) || (pwd.length == 6) ) {
        terms['length'] = true;
      } else {
        terms['length'] = false;
      }
      //проверка на наличие латинских строчных букв
      if ((/[a-z]/).test(pwd)) {
        terms['az'] = true;
      } else {
        terms['az'] = false;
      }
      //проверка на наличие латинских заглавных букв
      if ((/[A-Z]/).test(pwd)) {
        terms['AZ'] = true;
      } else {
        terms['AZ'] = false;
      }
      //проверка на наличие символов
      if ((/[!#$%^\*&\@\?\+]/).test(pwd)) {
        terms['symbols'] = true;
      } else {
        terms['symbols'] = false;
      }
      //проверка на наличие цифр
      if ((/[0-9]/).test(pwd)) {
        terms['int'] = true;
      } else {
        terms['int'] = false;
      }
      for (key in terms) {
        if (terms[key]) {
          protect++;
        }
      }
      //проверка на наличие недопустимых символов
      if ((/^[a-zA-Z0-9!#$%_^\*&\@\?\+]*$/).test(pwd)) {
        error = false;
      } else {
        error = true;
     }
      if (error) {
       // при наличии недопустимых символов
       // текст в блоке .help-block
       // меняет цвет на красный - предупреждение
        $('.help-block').css({'color':'#A21F1F'});
      } else {
        if ( protect == 5 ) {
         // при наличии 4-х видов символов - пароль неплохой
         $('#protect').css({"color": "#0C640C"});
         $('#protect').html("надежный пароль");
        } else if ( protect == 4 ) {
         // при наличии всех видов символов - пароль надежный
         $('#protect').css({"color": "#FFFF00"});
         $('#protect').html("неплохой пароль");
        } else {
         $('#protect').css({"color": "#A21F1F"});
         $('#protect').html("ненадежный пароль");
        }
        $('.help-block').css({'color':'#000000'});
      }
     });
    });
